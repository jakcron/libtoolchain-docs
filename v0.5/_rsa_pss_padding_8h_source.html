208 {
209 HashFunction hash;
210 std::array<byte_t, HashSize> mask;
211 tc::bn::be32<uint32_t> beRoundNum;
212
213 for (size_t round_idx = 0, round_num = (dst_size + HashSize - 1) / HashSize; round_idx < round_num; round_idx++)
214 {
215 hash.initialize();
216
217 // update using src data
218 hash.update(src, src_size);
219
220 // update using big endian round num
221 beRoundNum.wrap((uint32_t)round_idx);
222 hash.update((byte_t*)&beRoundNum, sizeof(tc::bn::be32<uint32_t>));
223
224 // get mask
225 hash.getHash(mask.data());
226
227 // merge mask and dst
228 size_t dst_pos = round_idx * HashSize;
229
230 for (size_t i = 0, len = std::min(dst_size - dst_pos, HashSize); i < len; i++)

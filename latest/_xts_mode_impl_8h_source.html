66 void encrypt(byte_t* dst, const byte_t* src, size_t size, uint64_t sector_number)
67 {
68 if (mState != State::Initialized) { return ; }
69 if (dst == nullptr) { throw tc::ArgumentNullException("XtsModeImpl::encrypt()", "dst was null."); }
70 if (src == nullptr) { throw tc::ArgumentNullException("XtsModeImpl::encrypt()", "src was null."); }
71 if (size == 0 || size % mSectorSize) { throw tc::ArgumentOutOfRangeException("XtsModeImpl::encrypt()", "size was not a multiple of the sector size."); }
72
73 auto block = std::array<byte_t, kBlockSize>();
74 auto dec_tweak = std::array<byte_t, kBlockSize>();
75 auto enc_tweak = std::array<byte_t, kBlockSize>();
76
77 // for ciphertext stealing
78 size_t sector_leftover = mSectorSize % kBlockSize;
79
80 // initialize tweak
81 set_tweak(dec_tweak.data(), sector_number);
82
83 // iterate through sectors
84 for (size_t sector_idx = 0, sector_num = (size / mSectorSize); sector_idx < sector_num; sector_idx++)
85 {
86 // encrypt tweak
87 mTweakCipher.encrypt(enc_tweak.data(), dec_tweak.data());
88
89 // process each block within a sector
90 for (size_t block_idx = 0, block_num = (mSectorSize / kBlockSize); block_idx < block_num; block_idx++)
91 {
92 const byte_t* src_block = src + (sector_idx * mSectorSize) + (block_idx * kBlockSize);
93 byte_t* dst_block = dst + (sector_idx * mSectorSize) + (block_idx * kBlockSize);
94
95 // block = src_block XOR enc_tweak
96 xor_block(block.data(), enc_tweak.data(), src_block);
97
98 // encrypt block
99 mCryptCipher.encrypt(block.data(), block.data());
100
101 // dst_block = enc_block XOR enc_tweak
102 xor_block(dst_block, block.data(), enc_tweak.data());
103
104 // Update encrypted tweak
105 galois_func(enc_tweak.data());
106 }
107
108 // cipher text stealing
109 if (sector_leftover > 0)
110 {
111 size_t block_idx = (mSectorSize / kBlockSize);
112 const byte_t* src_block = src + (sector_idx * mSectorSize) + (block_idx * kBlockSize);
113 byte_t* prev_dst_block = dst + (sector_idx * mSectorSize) + ((block_idx - 1) * kBlockSize);
114 byte_t* dst_block = dst + (sector_idx * mSectorSize) + (block_idx * kBlockSize);
115
116 for (size_t j = 0; j < sector_leftover; j++)
117 {
118 // block [0, sector_leftover) = src_block [0, sector_leftover) ^ enc_tweak[0, sector_leftover)
119 block[j] = src_block[j] ^ enc_tweak[j];
120
121 // dst_block [0, sector_leftover) = prev_dst_block [0, sector_leftover)
122 dst_block[j] = prev_dst_block[j];
123 }
124
125 for (size_t j = sector_leftover; j < kBlockSize; j++)
126 {
127 // block [sector_leftover, kBlockSize) = prev_dst_block[sector_leftover, kBlockSize) ^ enc_tweak[sector_leftover, kBlockSize)
128 block[j] = prev_dst_block[j] ^ enc_tweak[j];
129 }
130
131 // encrypt block
132 mCryptCipher.encrypt(block.data(), block.data());
133
134 // prev_dst_block = enc_block XOR enc_tweak
135 xor_block(prev_dst_block, block.data(), enc_tweak.data());
136 }
137
138 // increment tweak
139 incr_tweak(dec_tweak.data(), 1);
140 }

267 inline void incr_tweak_be(byte_t* tweak, uint64_t incr)
268 {
269 tc::bn::be64<uint64_t>* tweak_words = (tc::bn::be64<uint64_t>*)tweak;
270
271 uint64_t carry = incr;
272 for (size_t i = 0; carry != 0 ; i = ((i + 1) % 2))
273 {
274 uint64_t word = tweak_words[1 - i].unwrap();
275 uint64_t remaining = std::numeric_limits<uint64_t>::max() - word;
276
277 if (remaining > carry)
278 {
279 tweak_words[1 - i].wrap(word + carry);
280 carry = 0;
281 }
282 else
283 {
284 tweak_words[1 - i].wrap(carry - remaining - 1);
285 carry = 1;
286 }
287 }
288 }

318 inline void galois_func(byte_t* tweak)
319 {
320 tc::bn::le64<uint64_t>* tweak_u64 = (tc::bn::le64<uint64_t>*)tweak;
321
322 uint64_t ra = ( tweak_u64[0].unwrap() << 1 ) ^ 0x0087 >> ( 8 - ( ( tweak_u64[1].unwrap() >> 63 ) << 3 ) );
323 uint64_t rb = ( tweak_u64[0].unwrap() >> 63 ) | ( tweak_u64[1].unwrap() << 1 );
324
325 tweak_u64[0].wrap(ra);
